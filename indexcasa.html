// Wait for the document to load document.addEventListener("DOMContentLoaded", function() { // Get the logo and lettering elements const logo = document.getElementById("logo"); const lettering = document.getElementById("lettering"); // Fade in the logo setTimeout(function() { logo.style.display = "block"; fadeIn(logo); }, 0); // After 5 seconds, stop fading the logo and fade in the lettering setTimeout(function() { clearInterval(logoFadeInterval); setTimeout(function() { lettering.style.display = "block"; fadeIn(lettering); }, 0); }, 5000); // After 3 seconds, redirect to homepage.html setTimeout(function() { window.location.href = "homepage.html"; }, 8000); }); // Function to fade in an element function fadeIn(element) { element.style.opacity = 0; var opacity = 0; var logoFadeInterval = setInterval(function() { if (opacity < 1) { opacity += 0.01; element.style.opacity = opacity; } else { clearInterval(logoFadeInterval); } }, 50); }
